function update_size(el) {
    var newsize = el.value;
    var asel = document.getElementById("size-arcsec");
    if (asel) {
        if (newsize == "") {
            asel.innerHTML = "";
        } else {
            newsize = parseFloat(newsize);
            asel.innerHTML = newsize/4;
        }
    }
}
function insertTerm(el) {
    var sbox = document.getElementById('pos');
    if (sbox) {
        if (typeof(el) == "string") {
            // strip off trailing 'r=0', which is irrelevant in this interface
            var i = el.indexOf(" r=0");
            if (i > 0) el = el.substring(0,i);
            sbox.value = el;
        } else {
            sbox.value = el.innerHTML;
        }
        sbox.focus();
        // find the submit button and click it
        var inputs = sbox.form.getElementsByTagName("input");
        for (i=0; i<inputs.length; i++) {
            if (inputs[i].type == "submit") {
                inputs[i].click();
                break;
            }
        }
    }
    return false;
}
function setCheckboxes(basename, vlist, values) {
    // set checked values for a bunch of form entries
    for (var i=0; i<vlist.length; i++) {
        var el = document.getElementById(basename+vlist[i]);
        if (el) el.checked = values[i];
    }
}
function selectMenuItem(menuname, item) {
    // select an item from a popup menu
    var el = document.getElementById(menuname);
    if (el) {
        for (var i=0; i < el.options.length; i++) {
            if (el.options[i].value == item) {
                el.options[i].selected = true;
                return;
            }
        }
    }
}
function setTextbox(boxname, value) {
    // set text box to given value
    var el = document.getElementById(boxname);
    if (el) el.value = value;
}
function clearForm() {
    // restore all form settings to initial defaults
    // search position = blank
    setTextbox("pos","");
    // select all filters
    var vlist, i, el, values;
    setCheckboxes("filter", ["color", "g", "r", "i", "z", "y"], [true,true,true,true,true,true]);
    // stack only
    setCheckboxes("filetype", ["stack", "warp"], [true,false]);
    // data only
    setCheckboxes("auxiliary", ["data", "mask", "wt", "exp", "expwt", "num"],
        [true,false,false,false,false,false]);
    // cutout size = 240 pixels = 1 arcmin
    setTextbox("cutout_size", "240");
    // display size is 0
    selectMenuItem("output_size", "0");
}
